  <script>

    const { createApp, ref, computed } = Vue;

    // 🔥🔥🔥 【超重要設定】 🔥🔥🔥 貼上你剛剛重新部署獲得的 GAS 網址

    const GAS_URL = 'https://script.google.com/macros/s/AKfycbw1Jr0NH6dLT7Hv4ZKsyHRBKBeRh_wEwRU_QDRgnHRoOdE2we0fQLjchCZHj9NTP0-5/exec';

    // 將風向角度轉換為方位文字的輔助函數

    const getWindDirection = (degree) => {

      if (degree >= 337.5 || degree < 22.5) return '北風';

      if (degree >= 22.5 && degree < 67.5) return '東北風';

      if (degree >= 67.5 && degree < 112.5) return '東風';

      if (degree >= 112.5 && degree < 157.5) return '東南風';

      if (degree >= 157.5 && degree < 202.5) return '南風';

      if (degree >= 202.5 && degree < 247.5) return '西南風';

      if (degree >= 247.5 && degree < 292.5) return '西風';

      if (degree >= 292.5 && degree < 337.5) return '西北風';

      return '未知';

    };

    createApp({

      setup() {

        const activeTab = ref('add');

        const loadingLocation = ref(false);

        const saving = ref(false);

        const currentData = ref(null);

        const notes = ref('');

        const photoBase64 = ref('');

        const logs = ref([]);

        const loadingHistory = ref(false);

        const searchTerm = ref('');

        const getLocationAndWeather = () => {

          if (!navigator.geolocation) { alert('瀏覽器不支援定位！'); return; }

          loadingLocation.value = true;

          navigator.geolocation.getCurrentPosition(

            async (position) => {

              const lat = position.coords.latitude;

              const lon = position.coords.longitude;

              try {

                // 1. 抓取地址

                const geoRes = await fetch(`https://api.bigdatacloud.net/data/reverse-geocode-client?latitude=${lat}&longitude=${lon}&localityLanguage=zh-tw`);

                const geoData = await geoRes.json();

                const address = geoData.locality || geoData.city || '未知沿海地區';

                // 2. 抓取陸地氣象 (增加 surface_pressure 和 wind_direction_10m)

                const weatherUrl = `https://api.open-meteo.com/v1/forecast?latitude=${lat}&longitude=${lon}&current=temperature_2m,relative_humidity_2m,surface_pressure,wind_speed_10m,wind_direction_10m&timezone=auto`;

                const weatherRes = await fetch(weatherUrl);

                const weatherData = await weatherRes.json();

               

                const wCurrent = weatherData.current;

               

                // 3. 抓取海象/潮汐資料

                let waveInfo = "無附近海象觀測站資料";

                try {

                  const marineUrl = `https://marine-api.open-meteo.com/v1/marine?latitude=${lat}&longitude=${lon}&current=wave_height,wave_direction,wave_period&timezone=auto`;

                  const marineRes = await fetch(marineUrl);

                  const marineData = await marineRes.json();

                  if (marineData.current && marineData.current.wave_height !== null) {

                    waveInfo = `浪高 ${marineData.current.wave_height}m | 週期 ${marineData.current.wave_period}s`;

                  }

                } catch (e) { console.log("無海象資料"); }

                currentData.value = {

                  lat, lon,

                  address: `${geoData.principalSubdivision || ''} ${address}`,

                  temperature: wCurrent.temperature_2m,

                  humidity: wCurrent.relative_humidity_2m,

                  pressure: wCurrent.surface_pressure, // 氣壓

                  windSpeed: wCurrent.wind_speed_10m,

                  windDirDeg: wCurrent.wind_direction_10m, // 用於旋轉圖示

                  windDirection: getWindDirection(wCurrent.wind_direction_10m), // 中文方位

                  tide: waveInfo,

                };

              } catch (error) {

                alert('抓取觀測資料失敗，請確認網路連線。');

              } finally {

                loadingLocation.value = false;

              }

            },

            (error) => {

              alert('無法取得定位，請確認已開啟手機 GPS 權限！');

              loadingLocation.value = false;

            },

            { enableHighAccuracy: true, timeout: 10000, maximumAge: 0 }

          );

        };

        const handlePhotoUpload = (e) => {

          const file = e.target.files[0];

          if (!file) return;

          const reader = new FileReader();

          reader.readAsDataURL(file);

          reader.onload = (event) => {

            const img = new Image();

            img.src = event.target.result;

            img.onload = () => {

              const canvas = document.createElement('canvas');

              const MAX_WIDTH = 800;

              let scaleSize = 1;

              if (img.width > MAX_WIDTH) { scaleSize = MAX_WIDTH / img.width; }

              canvas.width = img.width * scaleSize;

              canvas.height = img.height * scaleSize;

              const ctx = canvas.getContext('2d');

              ctx.drawImage(img, 0, 0, canvas.width, canvas.height);

              photoBase64.value = canvas.toDataURL('image/jpeg', 0.6);

            };

          };

        };

        const saveLog = async () => {

          if (!currentData.value && !notes.value && !photoBase64.value) {

            alert('請至少填寫筆記或進行定位喔！'); return;

          }

          if (GAS_URL === '請在這裡貼上你的_GAS_網址') {

            alert('請先在程式碼中設定 GAS_URL！'); return;

          }

          saving.value = true;

          const payload = {

            dateString: new Date().toLocaleDateString('zh-TW', { year: 'numeric', month: '2-digit', day: '2-digit' }),

            notes: notes.value,

            photoBase64: photoBase64.value,

            ...(currentData.value || {})

          };

          try {

            await fetch(GAS_URL, {

              method: 'POST',

              body: JSON.stringify(payload),

              headers: { 'Content-Type': 'text/plain;charset=utf-8' }

            });

            alert('紀錄成功發送至 Google 試算表！🎉');

            currentData.value = null;

            notes.value = '';

            photoBase64.value = '';

            await fetchLogs();

            activeTab.value = 'history';

          } catch (error) {

            alert('儲存失敗，請檢查網路。');

          } finally {

            saving.value = false;

          }

        };

        const fetchLogs = async () => {

          if (GAS_URL === '請在這裡貼上你的_GAS_網址') return;

          loadingHistory.value = true;

          try {

            const response = await fetch(GAS_URL);

            const data = await response.json();

            logs.value = data.reverse();

          } catch (error) {

            alert('讀取歷史紀錄失敗。');

          } finally {

            loadingHistory.value = false;

          }

        };

        const filteredLogs = computed(() => {

          return logs.value.filter(log => {

            return searchTerm.value === '' ||

              (log.notes && log.notes.toLowerCase().includes(searchTerm.value.toLowerCase())) ||

              (log.address && log.address.toLowerCase().includes(searchTerm.value.toLowerCase()));

          });

        });

        return {

          activeTab, loadingLocation, saving, currentData, notes, photoBase64,

          logs, loadingHistory, searchTerm, filteredLogs,

          getLocationAndWeather, handlePhotoUpload, saveLog, fetchLogs

        };

      }

    }).mount('#app');
