   const { createApp, ref, computed } = Vue;

   // 🔥🔥🔥 【超重要設定】 🔥🔥🔥

   // 請將下方字串替換成你發佈 Google Apps Script 後取得的「網頁應用程式 URL」

   const GAS_URL = '請在這裡貼上你的_GAS_網址';

   createApp({

     setup() {

       const activeTab = ref('add');

     

       // 表單狀態

       const loadingLocation = ref(false);

       const saving = ref(false);

       const currentData = ref(null);

       const notes = ref('');

       const photoBase64 = ref('');

     

       // 歷史與搜尋狀態

       const logs = ref([]);

       const loadingHistory = ref(false);

       const searchTerm = ref('');

       const searchDate = ref('');

       // 1. 取得定位與氣象

       const getLocationAndWeather = () => {

         if (!navigator.geolocation) {

           alert('你的瀏覽器不支援地理定位功能喔！'); return;

         }

         loadingLocation.value = true;

         navigator.geolocation.getCurrentPosition(

           async (position) => {

             const lat = position.coords.latitude;

             const lon = position.coords.longitude;

             try {

               // 抓取地址

               const geoRes = await fetch(`https://api.bigdatacloud.net/data/reverse-geocode-client?latitude=${lat}&longitude=${lon}&localityLanguage=zh-tw`);

               const geoData = await geoRes.json();

               const address = geoData.locality || geoData.city || '未知沿海地區';

               // 抓取氣象

               const weatherRes = await fetch(`https://api.open-meteo.com/v1/forecast?latitude=${lat}&longitude=${lon}&current=temperature_2m,relative_humidity_2m,wind_speed_10m&timezone=auto`);

               const weatherData = await weatherRes.json();

             

               // 抓取浪高

               let waveInfo = "無觀測資料";

               try {

                 const marineRes = await fetch(`https://marine-api.open-meteo.com/v1/marine?latitude=${lat}&longitude=${lon}&current=wave_height&timezone=auto`);

                 const marineData = await marineRes.json();

                 if (marineData.current && marineData.current.wave_height !== null) {

                   waveInfo = `浪高 ${marineData.current.wave_height}m`;

                 }

               } catch (e) { console.log("無海象資料"); }

               currentData.value = {

                 lat, lon,

                 address: `${geoData.principalSubdivision || ''} ${address}`,

                 temperature: weatherData.current.temperature_2m,

                 humidity: weatherData.current.relative_humidity_2m,

                 windSpeed: weatherData.current.wind_speed_10m,

                 tide: waveInfo,

               };

             } catch (error) {

               alert('抓取氣象資料失敗，請確認網路連線。');

             } finally {

               loadingLocation.value = false;

             }

           },

           (error) => {

             alert('無法取得定位，請確認手機與瀏覽器已開啟 GPS 權限！');

             loadingLocation.value = false;

           },

           { enableHighAccuracy: true, timeout: 10000, maximumAge: 0 }

         );

       };

       // 2. 處理照片上傳與壓縮 (避免傳送太大的檔案給 GAS)

       const handlePhotoUpload = (e) => {

         const file = e.target.files[0];

         if (!file) return;

         const reader = new FileReader();

         reader.readAsDataURL(file);

         reader.onload = (event) => {

           const img = new Image();

           img.src = event.target.result;

           img.onload = () => {

             const canvas = document.createElement('canvas');

             const MAX_WIDTH = 800; // 最大寬度

             let scaleSize = 1;

             if (img.width > MAX_WIDTH) { scaleSize = MAX_WIDTH / img.width; }

           

             canvas.width = img.width * scaleSize;

             canvas.height = img.height * scaleSize;

             const ctx = canvas.getContext('2d');

             ctx.drawImage(img, 0, 0, canvas.width, canvas.height);

           

             // 壓縮並轉為 Base64

             photoBase64.value = canvas.toDataURL('image/jpeg', 0.6);

           };

         };

       };

       // 3. 儲存資料至 Google 試算表

       const saveLog = async () => {

         if (!currentData.value && !notes.value && !photoBase64.value) {

           alert('請至少填寫一些紀錄或進行定位喔！'); return;

         }

         if (GAS_URL === '請在這裡貼上你的_GAS_網址') {

           alert('你還沒有設定 GAS 網址喔！請先修改原始碼。'); return;

         }

         saving.value = true;

       

         // 準備要傳送的資料

         const payload = {

           dateString: new Date().toLocaleDateString('zh-TW', { year: 'numeric', month: '2-digit', day: '2-digit' }),

           notes: notes.value,

           photoBase64: photoBase64.value,

           ...(currentData.value || {})

         };

         try {

           await fetch(GAS_URL, {

             method: 'POST',

             body: JSON.stringify(payload),

             // 使用 text/plain 避免觸發 CORS 預檢請求 (GAS 不支援 OPTIONS 請求)

             headers: { 'Content-Type': 'text/plain;charset=utf-8' }

           });

           alert('紀錄成功發送至你的 Google 試算表！🎉');

           currentData.value = null;

           notes.value = '';

           photoBase64.value = '';

         

           // 儲存後自動刷新歷史紀錄並切換頁籤

           await fetchLogs();

           activeTab.value = 'history';

         

         } catch (error) {

           console.error("Save Error:", error);

           alert('儲存發生異常，請檢查網路連線。');

         } finally {

           saving.value = false;

         }

       };

       // 4. 讀取 Google 試算表的歷史紀錄

       const fetchLogs = async () => {

         if (GAS_URL === '請在這裡貼上你的_GAS_網址') return;

       

         loadingHistory.value = true;

         try {

           const response = await fetch(GAS_URL);

           const data = await response.json();

           // 反轉陣列，讓最新紀錄排在最上面

           logs.value = data.reverse();

         } catch (error) {

           console.error("Fetch Error:", error);

           alert('讀取歷史紀錄失敗。');

         } finally {

           loadingHistory.value = false;

         }

       };

       // 5. 搜尋過濾功能

       const filteredLogs = computed(() => {

         return logs.value.filter(log => {

           const matchKeyword = searchTerm.value === '' ||

             (log.notes && log.notes.toLowerCase().includes(searchTerm.value.toLowerCase())) ||

             (log.address && log.address.toLowerCase().includes(searchTerm.value.toLowerCase()));

           

           const formattedSearchDate = searchDate.value.replace(/-/g, '/');

           const matchDate = searchDate.value === '' || (log.dateString && log.dateString === formattedSearchDate);

         

           return matchKeyword && matchDate;

         });

       });

       return {

         activeTab, loadingLocation, saving, currentData, notes, photoBase64,

         logs, loadingHistory, searchTerm, searchDate, filteredLogs,

         getLocationAndWeather, handlePhotoUpload, saveLog, fetchLogs

       };

     }

   }).mount('#app');

 </script>
